if(POLICY CMP0015)
   cmake_policy(SET CMP0015 NEW)
endif(POLICY CMP0015)


if( EXISTS $ENV{TRITON_DIR} )
	set (TRITON_DIR $ENV{TRITON_DIR} )
elif( EXISTS $ENV{TRITON_PATH} )
	set (TRITON_DIR $ENV{TRITON_PATH} )	
else()
	set (TRITON_DIR ${CMAKE_SOURCE_DIR}/../..)
endif()

message(STATUS "TRITON_DIR environment variable is " ${TRITON_DIR})
message(STATUS "CMAKE_CURRENT_SOURCE_DIR is " ${CMAKE_CURRENT_SOURCE_DIR})
message(STATUS "CMAKE_SOURCE_DIR is " ${CMAKE_SOURCE_DIR})
message(STATUS "CMAKE_CURRENT_LIST_DIR is " ${CMAKE_CURRENT_LIST_DIR})
message(STATUS "CMAKE_CURRENT_DIR is " ${CMAKE_CURRENT_DIR})
message(STATUS "CMAKE_CURRENT_DIRECTORY is " ${CMAKE_CURRENT_DIRECTORY})

set(CMAKE_MODULE_PATH "${CMAKE_CURRENT_SOURCE_DIR}/CMakeModules;${CMAKE_MODULE_PATH}")
find_package(Triton REQUIRED)

set(src_dir ${CMAKE_CURRENT_SOURCE_DIR}/src)

file(GLOB cpp_files ${src_dir}/*.cpp)
file(GLOB h_files ${src_dir}/*.h)

#this file is automatically generated 
SET(TARGET_SRC ${cpp_files} ${h_files})

#### end var setup  ###
SETUP_EXAMPLE(TritonOcean)

message(STATUS "TRITON_LIBRARY_DEBUG is " ${TRITON_LIBRARY_DEBUG})
message(STATUS "TRITON_LIBRARY_RELEASE is " ${TRITON_LIBRARY_RELEASE})
message(STATUS "src_dir is " ${src_dir})
message(STATUS "cpp_files is " ${cpp_files})
message(STATUS "h_files is " ${h_files})
message(STATUS "TRITON_INCLUDE_DIR is " ${TRITON_INCLUDE_DIR})

target_link_libraries(example_TritonOcean debug ${TRITON_LIBRARY_DEBUG} optimized ${TRITON_LIBRARY_RELEASE})
target_include_directories(example_TritonOcean PUBLIC ${TRITON_INCLUDE_DIR})
